Can create a full CRUDable REST API that follows conventions.


import express from "express";
const app = express();
app.use(express.json());

let users = [];

app.get("/users", (req, res) => res.json(users));
app.post("/users", (req, res) => {
  const user = req.body;
  users.push(user);
  res.status(201).json(user);
});
app.put("/users/:id", (req, res) => {
  const { id } = req.params;
  users[id] = req.body;
  res.json(users[id]);
});
app.delete("/users/:id", (req, res) => {
  const { id } = req.params;
  users.splice(id, 1);
  res.status(204).send();
});

const PORT = 8080;
app.listen(PORT, (error) => {
    if (error) {
        console.log("Error starting the server", error);
        return;
    }
    console.log("Server is running on port", PORT);
});

Can get the Date in JavaScript.


// Get the current date and time
const now = new Date();

// Get parts of the date
const year = now.getFullYear();
const month = now.getMonth() + 1;
const day = now.getDate();

console.log(`Today's date is ${day}/${month}/${year}`);

Can write a fetch request without hesitation.


// GET request
fetch("https://api.example.com/users")
  .then(response => response.json())
  .then(data => console.log(data));

// POST request
fetch("https://api.example.com/users", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({ name: "Tobias", age: 25 })
})
  .then(response => response.json())
  .then(data => console.log("User created:", data))
  .catch(error => console.error("Error:", error));
